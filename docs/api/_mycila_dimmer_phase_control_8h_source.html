50 bool setDutyCycle(float dutyCycle) override {
51 // Apply limit and save the wanted duty cycle.
52 // It will only be applied when dimmer will be on.
53 _dutyCycle = _contrain(dutyCycle, 0, _dutyCycleLimit);
54
55 const float mapped = getDutyCycleMapped();
56
57 if (_powerLUTEnabled) {
58 if (mapped == 0) {
59 _dutyCycleFire = 0.0f;
60 } else if (mapped == 1) {
61 _dutyCycleFire = 1.0f;
62 } else {
63 _dutyCycleFire = _semiPeriod > 0 ? (1.0f - static_cast<float>(_lookupFiringDelay(mapped)) / static_cast<float>(_semiPeriod)) : mapped;
64 }
65 } else {
66 _dutyCycleFire = mapped;
67 }
68

76 float getPowerRatio() const override {
77 if (_powerLUTEnabled) {
78 // With LUT enabled, the mapped duty cycle IS the linearized power ratio
79 return getDutyCycleMapped();
80 } else {
81 // Without LUT, we have a linear firing time.
82 // We calculate real Power Ratio from Phase Angle using the physical formula:
83 // P_ratio = d - sin(2 * pi * d) / (2 * pi)
84 const float duty = getDutyCycleFire();
85 // Optimization: use multiplication by inverse instead of division
86 return duty - sinf(2.0f * M_PI * duty) * (1.0f / (2.0f * M_PI));
87 }

107 bool _calculateDimmerHarmonics(float* array, size_t n) const override {
108 // getDutyCycleFire() returns the conduction angle normalized (0-1)
109 // Convert to firing angle: α = π × (1 - conduction)
110 // At 50% power: α ≈ 90° (π/2), which gives maximum harmonics
111 const float firingAngle = M_PI * (1.0f - _dutyCycleFire);
112
113 // Calculate RMS of fundamental component (reference)
114 // Formula from Thierry Lequeu: I1_rms = (1/π) × √[2(π - α + ½sin(2α))]
115 const float sin_2a = sinf(2.0f * firingAngle);
116 const float i1_rms = sqrtf((2.0f / M_PI) * (M_PI - firingAngle + 0.5f * sin_2a));
117
118 if (i1_rms <= 0.001f)
119 return false;
120
121 array[0] = 100.0f; // H1 (fundamental) = 100% reference
122
123 // Pre-compute scale factor for efficiency
124 const float scale_factor = (2.0f / M_PI) * 0.70710678f * 100.0f / i1_rms;
125
126 // Calculate odd harmonics (H3, H5, H7, ...)
127 // Formula for phase-controlled resistive loads (IEEE standard):
128 // Hn = (2/π√2) × |cos((n-1)α)/(n-1) - cos((n+1)α)/(n+1)| / I1_rms × 100%
129 // This gives the correct harmonic magnitudes relative to the fundamental
130 for (size_t i = 1; i < n; i++) {
131 const float n_f = static_cast<float>(2 * i + 1); // 3, 5, 7, 9, ...
132 const float n_minus_1 = n_f - 1.0f;
133 const float n_plus_1 = n_f + 1.0f;
134
135 // Compute Fourier coefficient
136 const float coeff = cosf(n_minus_1 * firingAngle) / n_minus_1 -
137 cosf(n_plus_1 * firingAngle) / n_plus_1;
138
139 // Convert to percentage of fundamental
140 array[i] = fabsf(coeff) * scale_factor;
141 }
142
143 return true;
144 }
