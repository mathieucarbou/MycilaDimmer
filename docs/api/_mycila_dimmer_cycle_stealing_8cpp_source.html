82bool Mycila::CycleStealingDimmer::_apply() {
83 if (!_enabled)
84 return false;
85 if (!_online || !_semiPeriod || _dutyCycleFire == 0) {
86 if (_running) {
87 if (fire_timer != nullptr) {
88 ESP_ERROR_CHECK(gptimer_set_alarm_action(fire_timer, nullptr));
89 }
90 _running = false;
91 }
92 } else {
93 if (!_running) {
94 // Start the firing timer with a period equal to the semi-period
95 // If a ZCD is there, the start of the timer will be synced by the ZCD signal to be just before the 0V crossing
96 // Otherwise, we rely on the clock accuracy considering that a ZC dimmer can only activate or deactivate at 0V crossing
97 gptimer_alarm_config_t fire_timer_alarm_cfg = {
98 .alarm_count = _semiPeriod,
99 .reload_count = 0,
100 .flags = {.auto_reload_on_alarm = true}};
101 if (fire_timer != nullptr) {
102 ESP_ERROR_CHECK(gptimer_set_raw_count(fire_timer, 0));
103 ESP_ERROR_CHECK(gptimer_set_alarm_action(fire_timer, &fire_timer_alarm_cfg));
104 }
105 _running = true;
106 }
107 }
108 return true;
109}

119// Timer ISR to be called as soon as a dimmer needs to be fired
120bool ARDUINO_ISR_ATTR Mycila::CycleStealingDimmer::_fireTimerISR(gptimer_handle_t timer, const gptimer_alarm_event_data_t* event, void* arg) {
121 // Prevent re-entry: if this ISR takes longer than the timer period,
122 // we must not allow concurrent execution which could cause race conditions
123 if (isr_running) {
124 // ISR is already running - skip this alarm to prevent re-entry
125 return false;
126 }
127 isr_running = true;
128
129 // get the time we spent looping and eventually waiting for the lock
130 uint64_t fire_timer_count_value;
131 if (inlined_gptimer_get_raw_count(fire_timer, &fire_timer_count_value) != ESP_OK) {
132 // failed to get the timer count: just ignore this event
133 isr_running = false;
134 return false;
135 }
136
137 // Note: if locking takes too long (more than semi-period), a new timer event may be triggered
138 // while we are still in this ISR, but it will be ignored by the isr_running guard
139
140#ifndef MYCILA_DIMMER_NO_LOCK
141 // lock since we need to iterate over the list of dimmers
142 portENTER_CRITICAL_SAFE(&dimmers_spinlock);
143#endif
144
145 // go through all registered dimmers and compute the next action
146 struct RegisteredDimmer* current = dimmers;
147 while (current != nullptr) {
148 if (current->alarm_count != UINT16_MAX) {
149 // this dimmer has not yet been fired (< UINT16_MAX)
150 if (current->alarm_count <= fire_timer_count_value) {
151 // timer alarm has reached this dimmer alarm => time to fire this dimmer
152 gpio_ll_set_level(&GPIO, current->dimmer->_pin, HIGH);
153 // reset the alarm count to indicate that this dimmer has been fired
154 current->alarm_count = UINT16_MAX;
155 } else {
156 // dimmer has to be fired later => keep the minimum time at which we have to fire a dimmer
157 if (current->alarm_count < fire_timer_alarm_cfg.alarm_count)
158 fire_timer_alarm_cfg.alarm_count = current->alarm_count;
159 }
160 }
161 current = current->next;
162 }
163
164#ifndef MYCILA_DIMMER_NO_LOCK
165 // unlock the list of dimmers
166 portEXIT_CRITICAL_SAFE(&dimmers_spinlock);
167#endif
168
169 isr_running = false;
170 return false;
171}

174void Mycila::CycleStealingDimmer::_registerDimmer(Mycila::CycleStealingDimmer* dimmer) {
175 if (dimmers == nullptr) {
176 ESP_LOGI(TAG, "Starting dimmer firing ISR");
177
178 gptimer_config_t timer_config;
179 timer_config.clk_src = GPTIMER_CLK_SRC_DEFAULT;
180 timer_config.direction = GPTIMER_COUNT_UP;
181 timer_config.resolution_hz = 1000000; // 1MHz resolution
182 timer_config.flags.intr_shared = true;
183 timer_config.intr_priority = 0;
184#if ESP_IDF_VERSION >= ESP_IDF_VERSION_VAL(5, 3, 0)
185 timer_config.flags.backup_before_sleep = false;
186#endif
187#if ESP_IDF_VERSION >= ESP_IDF_VERSION_VAL(5, 4, 0)
188 timer_config.flags.allow_pd = false;
189#endif
190
191 ESP_ERROR_CHECK(gptimer_new_timer(&timer_config, &fire_timer));
192 gptimer_event_callbacks_t callbacks_config;
193 callbacks_config.on_alarm = _fireTimerISR;
194 ESP_ERROR_CHECK(gptimer_register_event_callbacks(fire_timer, &callbacks_config, nullptr));
195 ESP_ERROR_CHECK(gptimer_enable(fire_timer));
196 ESP_ERROR_CHECK(gptimer_start(fire_timer));
197 }
198
199 ESP_LOGD(TAG, "Register new dimmer %p on pin %d", dimmer, dimmer->getPin());
200
201#ifndef MYCILA_DIMMER_NO_LOCK
202 portENTER_CRITICAL_SAFE(&dimmers_spinlock);
203#endif
204
205 if (dimmers == nullptr) {
206 dimmers = new RegisteredDimmer();
207 dimmers->dimmer = dimmer;
208 } else {
209 struct RegisteredDimmer* first = new RegisteredDimmer();
210 first->next = dimmers;
211 dimmers->prev = first;
212 dimmers = first;
213 }
214
215#ifndef MYCILA_DIMMER_NO_LOCK
216 portEXIT_CRITICAL_SAFE(&dimmers_spinlock);
217#endif
218}

220// remove a dimmer from the list of managed dimmers
221void Mycila::CycleStealingDimmer::_unregisterDimmer(Mycila::CycleStealingDimmer* dimmer) {
222 ESP_LOGD(TAG, "Unregister dimmer %p on pin %d", dimmer, dimmer->getPin());
223
224#ifndef MYCILA_DIMMER_NO_LOCK
225 portENTER_CRITICAL_SAFE(&dimmers_spinlock);
226#endif
227
228 struct RegisteredDimmer* current = dimmers;
229 while (current != nullptr) {
230 if (current->dimmer == dimmer) {
231 if (current->prev != nullptr) {
232 current->prev->next = current->next;
233 } else {
234 dimmers = current->next;
235 }
236 if (current->next != nullptr) {
237 current->next->prev = current->prev;
238 }
239 delete current;
240 break;
241 }
242 current = current->next;
243 }
244
245#ifndef MYCILA_DIMMER_NO_LOCK
246 portEXIT_CRITICAL_SAFE(&dimmers_spinlock);
247#endif
248
249 if (dimmers == nullptr) {
250 ESP_LOGI(TAG, "Stopping dimmer firing ISR");
251 gptimer_stop(fire_timer); // might be already stopped
252 ESP_ERROR_CHECK(gptimer_disable(fire_timer));
253 ESP_ERROR_CHECK(gptimer_del_timer(fire_timer));
254 fire_timer = nullptr;
255 }
