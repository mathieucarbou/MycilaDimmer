82// Timer ISR to be called as soon as a dimmer needs to be fired
83bool ARDUINO_ISR_ATTR Mycila::CycleStealingDimmer::_fireTimerISR(gptimer_handle_t timer, const gptimer_alarm_event_data_t* event, void* arg) {
84 // Prevent re-entry: if this ISR takes longer than the timer period,
85 // we must not allow concurrent execution which could cause race conditions
86 if (inside_isr) {
87 // ISR is already running - skip this alarm to prevent re-entry
88 return false;
89 }
90 inside_isr = true;
91
92 // get the time we spent looping and eventually waiting for the lock
93 uint64_t fire_timer_count_value;
94 if (inlined_gptimer_get_raw_count(fire_timer, &fire_timer_count_value) != ESP_OK) {
95 // failed to get the timer count: just ignore this event
96 inside_isr = false;
97 return false;
98 }
99
100 // Note: if locking takes too long (more than semi-period), a new timer event may be triggered
101 // while we are still in this ISR, but it will be ignored by the inside_isr guard
102
103#ifndef MYCILA_DIMMER_NO_LOCK
104 // lock since we need to iterate over the list of dimmers
105 portENTER_CRITICAL_SAFE(&dimmers_spinlock);
106#endif
107
108 // Semi-period cycle stealing with balanced control
109 // Each semi-period (10ms for 50Hz), we decide whether to conduct or not
110 // To avoid DC components, we must balance positive and negative half-cycles
111
112 struct RegisteredDimmer* current = dimmers;
113 while (current != nullptr) {
114 CycleStealingDimmer* dimmer = current->dimmer;
115 float dutyCycle = dimmer->getDutyCycleFire();
116
117 // Full power: always conduct
118 if (dutyCycle >= 1.0f) {
119 gpio_ll_set_level(&GPIO, dimmer->_pin, HIGH);
120 current->semi_period_odd = !current->semi_period_odd;
121 current = current->next;
122 continue;
123 }
124
125 // Zero power: never conduct
126 if (dutyCycle <= 0.0f) {
127 gpio_ll_set_level(&GPIO, dimmer->_pin, LOW);
128 current->semi_period_odd = !current->semi_period_odd;
129 current = current->next;
130 continue;
131 }
132
133 // Cycle stealing algorithm with DC balance
134 // Sliding window approach (Bresenham) with polarity balancing
135
136 // Accumulate the energy deficit
137 current->density_error += dutyCycle;
138
139 bool should_conduct = false;
140
141 // Check if we have enough accumulated error to fire a pulse
142 if (current->density_error >= 1.0f) {
143 // We want to fire. Check DC balance constraints.
144 // semi_period_odd: True (Odd/Positive), False (Even/Negative)
145 // dc_balance: 0 (Balanced), >0 (Excess Positive), <0 (Excess Negative)
146 // Optimization: We define Odd as Positive (+1) and Even as Negative (-1)
147 int8_t phase_val = current->semi_period_odd ? 1 : -1;
148
149 // Rule:
150 // 1. If balanced (0), we can fire. We will create a debt.
151 // 2. If unbalanced, we can ONLY fire if it reduces the imbalance (opposite sign).
152
153 bool helps_balance = (current->dc_balance == 0) ||
154 (current->dc_balance > 0 && phase_val < 0) ||
155 (current->dc_balance < 0 && phase_val > 0);
156
157 if (helps_balance) {
158 should_conduct = true;
159 current->dc_balance += phase_val;
160 current->density_error -= 1.0f;
161 } else {
162 // We need to fire for power, but it would worsen the DC imbalance.
163 // Wait for the next semi-period (which will have opposite polarity).
164 should_conduct = false;
165 }
166 }
167
168 // Apply the decision
169 gpio_ll_set_level(&GPIO, dimmer->_pin, should_conduct ? HIGH : LOW);
170 current->semi_period_odd = !current->semi_period_odd;
171
172 current = current->next;
173 }
174
175#ifndef MYCILA_DIMMER_NO_LOCK
176 // unlock the list of dimmers
177 portEXIT_CRITICAL_SAFE(&dimmers_spinlock);
178#endif
179
180 inside_isr = false;
181 return false;
182}

185void Mycila::CycleStealingDimmer::_registerDimmer(Mycila::CycleStealingDimmer* dimmer) {
186 if (fire_timer == nullptr) {
187 ESP_LOGI(TAG, "Starting dimmer firing ISR");
188
189 gptimer_config_t timer_config;
190 timer_config.clk_src = GPTIMER_CLK_SRC_DEFAULT;
191 timer_config.direction = GPTIMER_COUNT_UP;
192 timer_config.resolution_hz = 1000000; // 1MHz resolution
193 timer_config.flags.intr_shared = true;
194 timer_config.intr_priority = 0;
195#if ESP_IDF_VERSION >= ESP_IDF_VERSION_VAL(5, 3, 0)
196 timer_config.flags.backup_before_sleep = false;
197#endif
198#if ESP_IDF_VERSION >= ESP_IDF_VERSION_VAL(5, 4, 0)
199 timer_config.flags.allow_pd = false;
200#endif
201
202 gptimer_event_callbacks_t callbacks_config;
203 callbacks_config.on_alarm = _fireTimerISR;
204
205 ESP_ERROR_CHECK(gptimer_new_timer(&timer_config, &fire_timer));
206 ESP_ERROR_CHECK(gptimer_register_event_callbacks(fire_timer, &callbacks_config, nullptr));
207 ESP_ERROR_CHECK(gptimer_enable(fire_timer));
208 ESP_ERROR_CHECK(gptimer_start(fire_timer));
209 }
210
211 ESP_LOGD(TAG, "Register new dimmer %p on pin %d", dimmer, dimmer->getPin());
212
213#ifndef MYCILA_DIMMER_NO_LOCK
214 portENTER_CRITICAL_SAFE(&dimmers_spinlock);
215#endif
216
217 if (dimmers == nullptr) {
218 dimmers = new RegisteredDimmer();
219 dimmers->dimmer = dimmer;
220 } else {
221 struct RegisteredDimmer* additional = new RegisteredDimmer();
222 additional->dimmer = dimmer;
223 additional->next = dimmers;
224 additional->prev = nullptr;
225 dimmers->prev = additional;
226 dimmers = additional;
227 }
228
229#ifndef MYCILA_DIMMER_NO_LOCK
230 portEXIT_CRITICAL_SAFE(&dimmers_spinlock);
231#endif
232}

234// remove a dimmer from the list of managed dimmers
235void Mycila::CycleStealingDimmer::_unregisterDimmer(Mycila::CycleStealingDimmer* dimmer) {
236 ESP_LOGD(TAG, "Unregister dimmer %p on pin %d", dimmer, dimmer->getPin());
237
238#ifndef MYCILA_DIMMER_NO_LOCK
239 portENTER_CRITICAL_SAFE(&dimmers_spinlock);
240#endif
241
242 struct RegisteredDimmer* current = dimmers;
243 while (current != nullptr) {
244 if (current->dimmer == dimmer) {
245 if (current->prev != nullptr) {
246 current->prev->next = current->next;
247 } else {
248 dimmers = current->next;
249 }
250 if (current->next != nullptr) {
251 current->next->prev = current->prev;
252 }
253 delete current;
254 break;
255 }
256 current = current->next;
257 }
258
259#ifndef MYCILA_DIMMER_NO_LOCK
260 portEXIT_CRITICAL_SAFE(&dimmers_spinlock);
261#endif
262
263 if (dimmers == nullptr && fire_timer != nullptr) {
264 ESP_LOGI(TAG, "Stopping dimmer firing ISR");
265 ESP_ERROR_CHECK(gptimer_set_alarm_action(fire_timer, nullptr));
266 ESP_ERROR_CHECK(gptimer_stop(fire_timer));
267 ESP_ERROR_CHECK(gptimer_disable(fire_timer));
268 ESP_ERROR_CHECK(gptimer_del_timer(fire_timer));
269 fire_timer = nullptr;
270 }
271}

273bool Mycila::CycleStealingDimmer::_apply() {
274 if (!_enabled)
275 return false;
276
277 // we have no semi-period (or we are disconnected) => make sure the alarm is disabled to not trigger ISR
278 if (_semiPeriod == 0 && alarm_set) {
279 if (fire_timer != nullptr) {
280 ESP_LOGD(TAG, "Disable firing timer alarm");
281 ESP_ERROR_CHECK(gptimer_set_alarm_action(fire_timer, nullptr));
282 }
283 alarm_set = 0;
284
285 // we have a semi-period to set and it's different from the current one => reset the alarm
286 } else if (_semiPeriod > 0 && alarm_set != _semiPeriod) {
287 if (fire_timer != nullptr) {
288 ESP_LOGD(TAG, "Enable firing timer alarm to %" PRIu16 " us", _semiPeriod);
289 gptimer_alarm_config_t fire_timer_alarm_cfg = {
290 .alarm_count = _semiPeriod,
291 .reload_count = 0,
292 .flags = {.auto_reload_on_alarm = true}};
293 ESP_ERROR_CHECK(gptimer_set_raw_count(fire_timer, 0));
294 ESP_ERROR_CHECK(gptimer_set_alarm_action(fire_timer, &fire_timer_alarm_cfg));
295 alarm_set = _semiPeriod;
296 } else {
297 alarm_set = 0;
298 }
299 }
300
301 return true;
302}
