39uint16_t Mycila::Dimmer::_lookupFiringDelay(float dutyCycle, uint16_t semiPeriod) {
40 uint32_t duty = dutyCycle * FIRING_DELAY_MAX;
41 uint32_t slot = duty * FIRING_DELAYS_SCALE + (FIRING_DELAYS_SCALE >> 1);
42 uint32_t index = slot >> 16;
43 uint32_t a = FIRING_DELAYS[index];
44 uint32_t b = FIRING_DELAYS[index + 1];
45 uint32_t delay = a - (((a - b) * (slot & 0xffff)) >> 16); // interpolate a b
46 return (delay * semiPeriod) >> 16; // scale to period
47}
