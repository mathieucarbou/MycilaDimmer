213 virtual float getPowerRatio() const {
214 // For a linear dimmer, the power ratio is directly proportional to the duty cycle
215 return getDutyCycleFire();
216 }

218 // Calculate harmonics based on dimmer firing angle for resistive loads
219 // array[0] = H1 (fundamental), array[1] = H3, array[2] = H5, array[3] = H7, etc.
220 // Only odd harmonics are calculated (even harmonics are negligible for symmetric dimmers)
221 // Returns true if harmonics were calculated, false if dimmer is not active
222 bool calculateHarmonics(float* array, size_t n) const {
223 if (array == nullptr || n == 0)
224 return false;
225
226 float duty = getDutyCycleFire();
227
228 // Check if dimmer is active and routing
229 if (duty <= 0.0f) {
230 for (size_t i = 0; i < n; i++) {
231 array[i] = 0.0f; // No power, no harmonics
232 }
233 return true;
234 }
235
236 if (duty >= 1.0f) {
237 array[0] = 100.0f; // H1 (fundamental) = 100% reference
238 for (size_t i = 1; i < n; i++) {
239 array[i] = 0.0f; // No harmonics at full power
240 }
241 return true;
242 }
243
244 // Initialize all values to NAN
245 for (size_t i = 0; i < n; i++) {
246 array[i] = NAN;
247 }
248
249 return _calculateDimmerHarmonics(array, n);
250 }

252 bool calculateMetrics(Metrics& metrics, float gridVoltage, float loadResistance) const {
253 if (!_enabled || loadResistance <= 0 || gridVoltage <= 0) {
254 return false;
255 }
256
257 const float powerRatio = getPowerRatio();
258
259 if (powerRatio <= 0) {
260 // no power
261 metrics.apparentPower = 0.0f;
262 metrics.current = 0.0f;
263 metrics.power = 0.0f;
264 metrics.powerFactor = NAN;
265 metrics.thdi = NAN;
266 metrics.voltage = 0.0f;
267 return true;
268 }
269
270 if (powerRatio >= 1) {
271 // full power
272 const float nominalPower = gridVoltage * gridVoltage / loadResistance;
273 metrics.apparentPower = nominalPower;
274 metrics.current = gridVoltage / loadResistance;
275 metrics.power = nominalPower;
276 metrics.powerFactor = 1.0f;
277 metrics.thdi = 0.0f;
278 metrics.voltage = gridVoltage;
279 return true;
280 }
281
282 const float nominalPower = gridVoltage * gridVoltage / loadResistance;
283
284 metrics.power = powerRatio * nominalPower;
285 metrics.powerFactor = std::sqrt(powerRatio);
286 metrics.voltage = metrics.powerFactor * gridVoltage;
287 metrics.current = metrics.voltage / loadResistance;
288 metrics.apparentPower = gridVoltage * metrics.current;
289
290 // THDi calculation for resistive load:
291 // PF = 1 / sqrt(1 + THDi^2) => THDi = sqrt(1/PF^2 - 1)
292 metrics.thdi = 100.0f * std::sqrt(1.0f / (metrics.powerFactor * metrics.powerFactor) - 1.0f);
293
294 return true;
295 }
